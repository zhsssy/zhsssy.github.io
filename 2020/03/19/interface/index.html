接口      

基本概念

interface 是一种类型，从它的定义可以看出来用了 type 关键字，准确的说 interface 是一种具有一组方法的类型。

如果一个类型实现了一个 interface 中所有方法，我们说类型实现了该 interface。go 没有显式的关键字用来实现 interface，只需要实现 interface 包含的方法即可。

empty interface

所有类型都实现了empty interface，因为任何一种类型至少实现了 0 个方法。空的 interface 没有方法，所以可以认为所有的类型都实现了 interface{}。如果定义一个函数参数是 interface{} 类型，这个函数应该可以接受任何类型作为它的参数。

duck typing

​ *”When I see a bird that walks like a duck and swims like a duck and quacks like a duck, I call that bird a duck.”*

一个对象有效的语义，不是由继承自特定的类或实现特定的接口，而是由”当前方法和属性的集合”决定 。

Go语言通过隐式接口机制实现了鸭子面向对象模型。

package main

import "fmt"

type Duck interface{
	Swim()
	Walk()
	Quacks()
}
type Bird struct {}
func (b *Bird) Swim() {
	fmt.Println("swim")
}
func (b *Bird) Walk() {
	fmt.Println("walk")
}
func (b *Bird) Quacks() {
	fmt.Println("quacks")
}
func main(){
	var duck Duck
	b := &Bird{}
	d = b
	duck.Swim()		
	duck.Walk()		
	duck.Quacks()	
}

运行后结果为

swim
walk
quacks

通过接口实现多态

package main

import "fmt"

type Programmer interface{
	Use()
	Learn()
}
type Gopher struct{}
func (g *Gopher) Use(){
	fmt.Println("gopher use go")
}
func (g *Gopher) Learn(){
	fmt.Println("gopher learn go")
}

type Pythonista struct{}
func (p *Pythonista) Use(){
	fmt.Println("pythonista use python")
}
func (p *Pythonista) Learn(){
	fmt.Println("pythonista learn python")
}
type Human struct {}

func (h *Human)Program(p Programmer){
	p.Use()
}

func Factory(name string) Programmer{
	switch name {
	case "gopher":
		return &Gopher{}
	case "pythonista":
		return &Pythonista{}
	default:
		panic("err")
	}
}

func main(){
	human := &Human{}
	p1  := Factory("gopher")
	p1.Learn()
	human.Program(p1)

	p2  := Factory("pythonista")
	p2.Learn()
	human.Program(p2)
}

gopher和pythonista两个对象并没有显式声明实现了接口programmer。实际上在编译器调用Program函数时，将gopher和pythonista对象转换成了Programmer类型。

Go 作为一种静态语言，通过接口实现了鸭子类型，实际上是 Go 的编译器在其中作了隐匿的转换工作。

源码

eface

type eface struct {
	_type *_type   //_type是对一种具体类型的描述,例如int,string
	data  unsafe.Pointer   //指向真实数据的指针
}

// Needs to be in sync with ../cmd/link/internal/ld/decodesym.go:/^func.commonsize,
// ../cmd/compile/internal/gc/reflect.go:/^func.dcommontype and
// ../reflect/type.go:/^type.rtype.
type _type struct {
    //类型大小
	size       uintptr //type size
	ptrdata    uintptr // size of memory prefix holding all pointers
    //hash值
	hash       uint32  //hash of type;avoids computation in hash table
	tflag      tflag	//// 类型的 flag，和反射相关
    //内存对齐相关
	align      uint8
	fieldalign uint8
    kind       uint8 // type mask bool,slice,struct...
	alg        *typeAlg
    //垃圾回收相关
	// gcdata stores the GC type data for the garbage collector.
	// If the KindGCProg bit is set in kind, gcdata is a GC program.
	// Otherwise it is a ptrmask bitmap. See mbitmap.go for details.
	gcdata    *byte
	str       nameOff //string form
	ptrToThis typeOff // type for pointer to this type, may be zero
}

iface

type iface struct {
	tab  *itab   //itab,类型信息
	data unsafe.Pointer
}

// layout of Itab known to compilers
// allocated in non-garbage-collected memory
// Needs to be in sync with
// ../cmd/compile/internal/gc/reflect.go:/^func.dumptypestructs.
type itab struct {
	inter  *interfacetype //interface type description   package path method
	_type  *_type         //origin type
	link   *itab  //hash表头指针
	hash   uint32 // copy of _type.hash. Used for type switches.
	bad    bool   // type does not implement interface
	inhash bool   // has this itab been added to hash?
	unused [2]byte
	fun    [1]uintptr //method具体实现
}
//包装了 _type 类型，_type 实际上是描述 Go 语言中各种数据类型的结构体。我们注意到，这里还包含一个 mhdr 字段，表示接口所定义的函数列表， pkgpath 记录定义了接口的包名
type interfacetype struct {
	typ     _type		
	pkgpath name
	mhdr    []imethod
}

type imethod struct {
	name nameOff
	ityp typeOff
}

接口的动态类型和动态值

例子

package main

import "fmt"

type Coder interface {
    code()
}

type Gopher struct {
    name string
}

func (g Gopher) code() {
    fmt.Printf("%s is coding\n", g.name)
}

func main() {
    var c Coder
    fmt.Println(c == nil)
    fmt.Printf("c: %T, %v\n", c, c)

    var g *Gopher
    fmt.Println(g == nil)

    c = g
    fmt.Println(c == nil)
    fmt.Printf("c: %T, %v\n", c, c)
}

输出

true
c: <nil>, <nil>
true
false
c: *main.Gopher, <nil>

在一开始动态类型和值都为nil，赋值后类型为*main.Gopher

空接口

nil 接口值既不保存值也不保存具体类型

只有当值和类型都为nil时，接口为空接口

例子

package main

import (
	"fmt"
	"reflect"
)

type TestnilStruct struct{}

func testnil1(a, b interface{}) bool {
	return a == b
}

func testnil2(a *TestnilStruct, b interface{}) bool {
	return a == b
}

func testnil3(a interface{}) bool {
	return a == nil
}

func testnil4(a *TestnilStruct) bool {
	return a == nil
}

func testnil5(a interface{}) bool {
	v := reflect.ValueOf(a)
	return v.IsNil() || !v.IsValid()		//判断返回值是否为nil，返回值是否有效
}

func main() {
	var a *TestnilStruct
	fmt.Println(testnil1(a, nil))
	fmt.Println(testnil2(a, nil))	
	fmt.Println(testnil3(a))		
	fmt.Println(testnil4(a))		
	fmt.Println(testnil5(a))		
}

结果

false
false
false
true
true

一个interface{}类型的变量包含了2个指针，一个指针指向值的类型，另外一个指针指向实际的值 对一个interface{}类型的nil变量来说，它的两个指针都是空。a传入后，指向的类型的指针不为空了，有类型的存在， 所以结果为false。 interface类型比较，两个指针都相等，才能相等。

类型断言

类型转换和类型断言的区别

类型转换前和后两者类型需要兼容

package main

import "fmt"

func main() {
	var i int = 9

	var f float64
	f = float64(i)
	fmt.Printf("%T, %v\n", f, f)

	f = 10.8
	a := int(f)
	fmt.Printf("%T, %v\n", a, a)

	s := []int(i)
}

int和float64之间的转换成功，但是在int转化为[]int时会报错

Cannot convert expression of type 'int' to type '[]int

当函数的形参为interface{}时，在函数中可以通过断言来判断interface的真实类型

安全类型断言

<目标类型值>,<布尔参数> := <表达式>.(目标类型)

非安全类型断言

<目标类型值> := <表达式>.(目标类型)

例子

package main

import "fmt"

type Student struct {
    Name string
    Age int
}

func main() {
    var i interface{} = new(Student)
    s := i.(Student)

    fmt.Println(s)
}

此时非安全断言会产生panic

此时的i类型为 *Student 而非 Student

func main() {
    var i interface{} = new(Student)
    s, ok := i.(Student)
    if ok {
        fmt.Println(s)
    }
}

采用安全类型断言即使失败也不会panic
